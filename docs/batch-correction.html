  # https://satijalab.org/seurat/articles/seurat5_integration

  so <- IntegrateLayers(so, method = HarmonyIntegration, orig.reduction = "pca", new.reduction = "harmony")

  so <- FindNeighbors(so, reduction = "harmony", dims = 1:num_dims)
